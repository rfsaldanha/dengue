race_ctrl <- control_race(parallel_over = "everything")

ref_simp_tuning_results <-

    "tune_race_anova",

    resamples = dc_ref_simp_split_folds,
    grid = 25,
    control = race_ctrl, 
    verbose = TRUE
  )
